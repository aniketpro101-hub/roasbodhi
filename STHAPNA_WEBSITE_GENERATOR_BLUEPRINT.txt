1. EXECUTIVE SUMMARY & HOW EVERYTHING WORKS
--------------------------------------------------------------------------------
Sthapna Instant Website Builder is an AI-powered SaaS engine designed for local businesses across India.
A user inputs their Business Name, City/Location, Industry Category, and a short description.

Within 60 seconds:
1. THE COPYWRITING ENGINE calls the Google Gemini API (gemini-2.5-flash / gemini-1.5-flash) to write localized headlines, taglines, about sections, 6 service descriptions with INR pricing, customer FAQs, and urgency promotional offers.
2. If the API key is missing or encounters a rate limit, the system seamlessly uses an INDUSTRY-AWARE FALLBACK ENGINE (`src/utils/templates.ts`), ensuring 0% failure rate (never blank).
3. THE LAYOUT ENGINE (`src/scripts/preview-layouts.js`) renders 10+ long-form sections complete with Google Fonts (Inter + Playfair Display), custom glassmorphism navigation, full-viewport 100vh hero headers with AI-generated 8K images, key metrics counters, interactive FAQ accordions, filterable category tabs, 5-star customer reviews, interactive Google Maps links, and a persistent pulsing WhatsApp float button.
4. THE RESPONSIVE PREVIEW SYSTEM allows testing in both Smartphone View and Full-Screen Browser Tab View.

import { getFallbackTemplateData } from '../../utils/templates';

export const prerender = false;

export const POST: APIRoute = async ({ request }) => {
  try {
    const body = await request.json();
    const { businessName, location, description, industry, phone, whatsapp } = body;
    const apiKey = import.meta.env.GEMINI_API_KEY || process.env.GEMINI_API_KEY;

    if (!apiKey) {
      const templateData = getFallbackTemplateData(businessName, location, industry, description, phone, whatsapp);
      return new Response(JSON.stringify(templateData), { status: 200, headers: { 'Content-Type': 'application/json' } });
    }

    const prompt = `Write high-converting website copy for local Indian small business: Name: ${businessName}, City: ${location}, Industry: ${industry}...`;
    // Gemini API fetch logic...
  } catch (err) {
    return new Response(JSON.stringify({ error: 'Server Error' }), { status: 500 });
  }
};
